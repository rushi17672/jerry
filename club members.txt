#include <iostream>
using namespace std;

struct Node {
    int prn;
    string name;
    Node* next;
};

Node* headA = nullptr;
Node* headB = nullptr;

// Add president
void addPresident(Node*& head, int prn, string name) {
    Node* newNode = new Node{prn, name, head};
    head = newNode;
}

// Add secretary
void addSecretary(Node*& head, int prn, string name) {
    Node* newNode = new Node{prn, name, nullptr};
    if (!head) {
        head = newNode;
        return;
    }
    Node* temp = head;
    while (temp->next) temp = temp->next;
    temp->next = newNode;
}

// Add member
void addmember(Node*& head, int prn, string name) {
    if (!head || !head->next) {
        cout << "Add President and Secretary first.\n";
        return;
    }
    Node* newNode = new Node{prn, name, nullptr};
    Node* temp = head;
    while (temp->next->next) temp = temp->next;
    newNode->next = temp->next;
    temp->next = newNode;
}

// Delete President
void deletePresident(Node*& head) {
    if (!head) {
        cout << "No President to delete.\n";
        return;
    }
    Node* temp = head;
    head = head->next;
    delete temp;
    cout << "President deleted.\n";
}

// Delete Secretary
void deleteSecretary(Node*& head) {
    if (!head || !head->next) {
        cout << "No secretary to delete.\n";
        return;
    }
    Node* temp = head;
    while (temp->next->next) temp = temp->next;
    delete temp->next;
    temp->next = nullptr;
    cout << "Secretary deleted.\n";
}

// Delete Member
void deleteMember(Node*& head, int prn) {
    if (!head || !head->next) {
        cout << "No members to delete.\n";
        return;
    }
    Node* temp = head;
    while (temp->next && temp->next->next) {
        if (temp->next->prn == prn) {
            Node* toDelete = temp->next;
            temp->next = temp->next->next;
            delete toDelete;
            cout << "Member deleted.\n";
            return;
        }
        temp = temp->next;
    }
    cout << "Member not found.\n";
}

// Display Members
void display(Node* head) {
    if (!head) {
        cout << "Club is empty.\n";
        return;
    }
    cout << "Club Members:\n";
    while (head) {
        cout << head->prn << " - " << head->name << endl;
        head = head->next;
    }
}

// Count Members
int countmembers(Node* head) {
    int count = 0;
    while (head) {
        count++;
        head = head->next;
    }
    return count;
}

// Concatenate lists
void concatenate(Node*& head1, Node* head2) {
    if (!head1) {
        head1 = head2;
        return;
    }
    Node* temp = head1;
    while (temp->next)
        temp = temp->next;
    temp->next = head2;
}

int main() {
    int choice, prn;
    string name;
    while (true) {
        cout << "--- Pinnacle Club Menu ---\n";
        cout << "1. Add President\n";
        cout << "2. Add Secretary\n";
        cout << "3. Add Member\n";
        cout << "4. Delete President\n";
        cout << "5. Delete Secretary\n";
        cout << "6. Delete Member\n";
        cout << "7. Display Members\n";
        cout << "8. Count Members\n";
        cout << "9. Concatenate Division B to A\n";
        cout << "10. Exit\n";
        cout << "Enter your choice: ";
        cin >> choice;

        switch (choice) {
        case 1:
            cout << "Enter PRN and Name of President: ";
            cin >> prn >> name;
            addPresident(headA, prn, name);
            break;
        case 2:
            cout << "Enter PRN and Name of Secretary: ";
            cin >> prn >> name;
            addSecretary(headA, prn, name);
            break;
        case 3:
            cout << "Enter PRN and Name of Member: ";
            cin >> prn >> name;
            addmember(headA, prn, name);
            break;
        case 4:
            deletePresident(headA);
            break;
        case 5:
            deleteSecretary(headA);
            break;
        case 6:
            cout << "Enter PRN of Member to delete: ";
            cin >> prn;
            deleteMember(headA, prn);
            break;
        case 7:
            display(headA);
            break;
        case 8:
            cout << "Total Members: " << countmembers(headA) << endl;
            break;
        case 9:
            cout << "Enter Division B members:\n";
            cout << "President PRN and Name: ";
            cin >> prn >> name;
            addPresident(headB, prn, name);
            cout << "Secretary PRN and Name: ";
            cin >> prn >> name;
            addSecretary(headB, prn, name);
            concatenate(headA, headB);
            cout << "Division B concatenated to Division A.\n";
            break;
        case 10:
            cout << "Exiting...\n";
            return 0;
        default:
            cout << "Invalid choice. Try again.\n";
        }
    }
    return 0;
}
